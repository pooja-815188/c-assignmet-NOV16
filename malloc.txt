/memory allocation using malloc

#include<stdio.h>
#include<stdlib.h>
int main()
{
    int n,i,*ptr;
    printf("enter a value\n");
    scanf("%d",&n);
    ptr = (int*)malloc(n*sizeof(int));
    printf("enter a values\n");
    for(i=0;i<n;i++)
    {
        scanf("%d",(ptr+i));
    }
    printf("entered values are\n");
    for(i=0;i<n;i++)
    {
        printf("%d\n",*(ptr+i));
    }
    free(ptr);
}
