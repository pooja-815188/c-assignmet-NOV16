#include<stdio.h>
#include<string.h>
#define MAX_LENGTH 100
void removeSp(char *str) 
{
    int i = 0, j = 0;
    while (str[i]!='\0') 
    {
        if (str[i]>='a' && str[i]<='z' || str[i]>='A' && str[i]<='Z') 
        {
            str[j++] = str[i];
        }
        i++;
    }
    str[j] = '\0';
}

int main()
{
    char str[MAX_LENGTH];
    printf("Enter a string: ");
    scanf("%s",str);
    removeSp(str);
    printf("string=%s\n", str);
}
